 - sample
 ```
 DEVICE="eth0"
 BOOTPROTO="static"
 MTU=1500
 ONBOOT=yes
 IPADDR=
 NETMASK=255.255.255.0
 GATEWAY=XX.XX.XX.254
 ```

 ```
 # service network restart
 ```
